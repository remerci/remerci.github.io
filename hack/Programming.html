    case "$1" in
        -w|--wget)
            NET_TOOL="wget"
            shift
            ;;
        -u|--url)
            HOSTS_URL="$2"
            shift 2
            ;;
        -h|-1|--help)
            usage
            exit 0
            shift
            ;;
        --)
            shift
            break
            ;;
    esac
done
